# 소드란?
   소드는 고랭의.strict 서브셋이라고 보면 됨. 일반적인 고랭 코드를 받아서 읽기 쉬운 C11 코드로 변환해 줌. 런타임은.ZERO, 가바지 컬렉션, 레퍼런스 카운팅, 숨겨진 할당도 없음. 
   스탠다드 라이브러리도 풍성하고, 네이티브 C와의 인터푀이도 가능함. 소드에서 C를 호출할 수도, C에서 소드를 호출할 수도 있음. CGO 같은 오버헤드는 없음. 
   고랭 툴링도 지원됨. 문법 하일라이팅, lsp, 린트, go test 등이 바로 가능함. 
   소드는 시스템 레벨 컨트롤을 원하는 고랭 개발자들을 위한 언어고, 고랭 같은 언어의 안전성, 구조, 툴링을 좋아하는 C 프로그래머들을 위한 언어임.
   # 소드 플레이그라운드
   소드 플레이그라운드에서는 Go 코드를 작성해서 바로 실행할 수 있음. Translate 버튼을 클릭하면 自动生成된 C 코드를 볼 수 있음. 
   ```go
   package main

   import (
       "solod.dev/so/conc"
       "solod.dev/so/mem"
       "solod.dev/so/sync/atomic"
   )

   // Account는 스레드 세이프한 돈 계좌임.
   type Account struct {
       Balance atomic.Int64
   }

   // Deposit는 돈을 추가함.
   func (a *Account) Deposit(amount int64) {
       a.Balance.Add(amount)
   }

   // pay는 공유 계좌에 10달러를 추가함.
   func pay(arg any) {
       acc := arg.(*Account)
       acc.Deposit(10)
   }

   func main() {
       var acc Account
       // 100번의 支付를 4개의 워커 스레드에서 처리함.
       opts := conc.PoolOptions{NumThreads: 4}
       pool := conc.NewPool(mem.System, opts)
       defer pool.Free()
       for range 100 {
           pool.Go(pay, &acc)
       }
       pool.Wait()
       println("balance is", acc.Balance.Load())
   }
   ```
